    const res = await fetch(process.env.NEXT_PUBLIC_GAS_URL ?? '', {
      method: 'POST',
      redirect: 'follow',
      headers: { 'Content-Type': 'text/plain;charset=utf-8' },
      body: JSON.stringify({
        formType: 'lead-capture',
        email,
        audienceCategory,
        name: name || undefined,
        role: role || undefined,
        timestamp: new Date().toISOString(),

    });
